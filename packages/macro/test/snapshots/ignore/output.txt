import { InputType as _InputType, Field as _Field, ObjectType as _ObjectType } from "type-graphql.macro/type-graphql";

@_ObjectType()
class File {
  @_Field(() => String)
  name: string;
  @_Field(() => Date)
  ctime: Date;
  @_Field(() => Date)
  mtime: Date;
  skipThis: boolean;
}

@_InputType()
class Input {
  @_Field(() => String)
  name: string;
  @_Field(() => String)
  password: string;
  skipThis: boolean;
}
